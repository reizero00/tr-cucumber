def initialize(build_name, product_ver, parent_routine = '', description: '')
	@name = build_name
	@description = description
	@prod_ver = product_ver

	unless parent_routine == '' # Execute block if a parent_routine was specified
		if parent_routine.class != TestrayRoutine
			@parent_routine_id = parent_routine if parent_routine.to_s.scan(/\D/).empty?
		else
			@parent_routine_id = parent_routine.id
			@parent_project_id = parent_routine.parent_project_id
		end
	end
end

def add_to_parent_routine(expected_status = 200, *parent_routine)
	begin # Try to get and store @parent_routine_id
		if parent_routine.class != TestrayRoutine
			unless parent_routine[0].to_s.scan(/\d/).empty? # If parent_routine given by ID (no letters)
				@parent_routine_id = parent_routine[0].to_i
				json_req('Get', "#{$testray_path}/routines/view.json", {}, expected_status: 200, uri_param: "?id=#{@parent_routine_id}")
				@parent_project_id = @hash_response['data']['testrayProjectId']
			end
		else
			@parent_routine_id = parent_routine.id
			@parent_project_id = parent_routine.parent_project_id
		end
	rescue NameError # if parent_routine not defined
		raise('Since @parent_routine_id is undefined, you must supply a parent_routine identifier in the call like: .add(parent_routine)')
	end

	json_req('Get', "#{$testray_path}/product_versions.json?testrayProjectId=#{@parent_project_id}", {}, expected_status: 200)
	begin
		@prod_ver_id = @hash_response['data'].find {|x| x['name'] == "#{@prod_ver}"}['testrayProductVersionId']
	rescue
		raise "Unable to store testrayProductVersionId because '#{@prod_ver}' was not found in the response."
	end

	req_data = {
		  testrayRoutineId: "#{@parent_routine_id}",
		  description: "#{@description}",
		  testrayProductVersionId: "#{@prod_ver_id}",
		  name: "#{@name}"
	}

	json_req('Post', "#{$testray_path}/builds.json", req_data, expected_status: expected_status)
	@id = get_hash_response('builds', @name) unless expected_status != 200
	self
end
