def initialize(parent_run, parent_case)
       if parent_run.class != TestrayRun
           @parent_run_id = parent_run if parent_run.to_s.scan(/\D/).empty?
       else
           @parent_run = parent_run
           @parent_run_id = parent_run.id
       end

       if parent_case.class != TestrayCase
           @parent_case_id = parent_case if parent_case.to_s.scan(/\D/).empty?
       else
           @parent_case = parent_case
           @parent_case_id = parent_case.id
           @name = parent_case.name
       end
end

def add_bulk_results(target_run, result_array_to_be_added, expected_status = 200)
	json_req('Post', "#{$testray_path}/case_results/bulk.json", {caseResultsJSONArrayString: "#{result_array_to_be_added}"}, expected_status: expected_status)

	@hash_response['data'].each do |result_response|
		target_run.link_result(TestrayCaseResult.new(target_run, $master_map['cases'][result_response['testrayCaseName']]).set_id(result_response['testrayCaseResultId']))
	end
	self
end

def add_to_parent_run(expected_status = 200, status: 0, comment: '', issues: '', assignee: '')
	request_data = {
		'testrayCaseId' => "#{@parent_case_id}",
		'testrayRunId' => "#{@parent_run_id}"
	}

	unless status == ''
		if status.class == String
			case status
				when 'blocked'
					request_data['status'] = 4
				when 'failed'
					request_data['status'] = 3
				when 'in progress'
					request_data['status'] = 1
				when 'passed'
					request_data['status'] = 2
				when 'retest'
					request_data['status'] = 5
				when 'test fix'
					request_data['status'] = 7
				when 'untested'
					request_data['status'] = 0
                   else
                       raise("Invalid status name: #{status}")
			end
		elsif status.class == Fixnum && status.between?(1,7)
			request_data['status'] = status
		end
	end

	unless comment == ''
		request_data['comment'] = CGI::escape("#{comment}")
	end

	unless issues == ''
		request_data['issues'] = "#{issues}"
       end

       unless assignee == ''
           request_data['assigned_user_id'] = get_user_ids(assignee).join(',')
       end

	json_req('Post', "#{$testray_path}/case_results/bulk.json", {caseResultsJSONArrayString: "#{[request_data]}"}, expected_status: expected_status)
	@id = @hash_response['data'][0]['testrayCaseResultId'] unless expected_status != 200

       @parent_run.link_result(self)
	self
end

def update(expected_status = 200, assigned_user_id: '', comment: '', issues: '', status: '', attachments: '')
    uri_params = "?id=#{@id}"
    uri_params << "&assignedUserId=#{assigned_user_id}" unless assigned_user_id == ''
    uri_params << "&comment=#{CGI::escape(comment)}" unless comment == ''
    uri_params << "&issues=#{issues}" unless issues == ''
    uri_params << "&status=#{status}" unless status == ''
    uri_params << "&attachments=#{attachments}" unless attachments== ''

    json_req('Get', "#{$testray_path}/case_results/update.json", {}, uri_param: uri_params, expected_status: expected_status)
end
