def delete(name_or_id, expected_status = 200)
	# Tests if project designated by name or ID
	if name_or_id.scan(/\d/).empty? # If project given by name
		name = name_or_id
		testray_get(name, 'projects')
		id = get_hash_response('projects', name)
		json_req('Post', "#{$testray_path}/projects/#{id}/delete.json", {}, expected_status: expected_status)
	else # If project given by ID
		id = name_or_id
		json_req('Post', "#{$testray_path}/projects/#{id}/delete.json", {}, expected_status: expected_status)
	end
end
