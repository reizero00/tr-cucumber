def initialize(routine_name, parent_project = '')
	@name = routine_name

	unless parent_project == '' # Execute block if a parent_project was specified
		if parent_project.class != TestrayProject
			if parent_project.to_s.scan(/\D/).empty? # If parent_project given by ID
				@parent_project_id = parent_project
			else # If parent_project given by name
				testray_get(parent_project, 'projects')
				@parent_project_id = get_hash_response('projects', parent_project)
			end
		else
			@parent_project_id = parent_project.id
		end
	end
end

def add_to_parent_project(expected_status = 200, *parent_project) #REMOVE_LATER: name not supplied because would have called TestrayRoutine.new first with a name
	unless defined?(@parent_project_id) # Execute block if @parent_project_id undefined
		begin # Try to get and store @parent_project_id
			if parent_project.class != TestrayProject
				unless parent_project.to_s.scan(/\d/).empty? # If parent_project given by ID
					@parent_project_id = parent_project
				end
			else
				@parent_project_id = parent_project.id
			end
		rescue NameError # if parent_project not defined
			raise('Since @parent_project_id is undefined, you must supply a parent_project identifier in the call like: .add(parent_project)')
		end
	end
	json_req('Post', "#{$testray_path}/routines.json", {testrayProjectId: "#{@parent_project_id}", name: "#{@name}"}, expected_status: expected_status)
	@id = get_hash_response('routines', @name) unless expected_status != 200
	self
end
