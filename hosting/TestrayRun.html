def initialize(factors, parent_build = '')
	@factors = factors

	unless parent_build == '' # Execute block if a parent_build was specified
		if parent_build.class != TestrayBuild
			@parent_build_id = parent_build if parent_build.to_s.scan(/\D/).empty?
		else
			@parent_build_id = parent_build.id
		end
	end
end

def add_to_parent_build(expected_status = 200, *parent_build)
	if @parent_build_id.nil?
		begin # Try to get and store @parent_build_id
			if parent_build.class != TestrayBuild
				unless parent_build.to_s.scan(/\d/).empty? # If parent_build given by ID (no letters)
					@parent_build_id = parent_build.to_i
				end
			else
				@parent_build_id = parent_build.id
			end
		rescue NameError # if parent_build not defined
			raise('Since @parent_build_id is undefined, you must supply a parent_build identifier in the call like: .add(parent_build)')
		end
	end

	request_data = {
		  testrayBuildId: "#{@parent_build_id}",
		  testrayFactors: "#{@factors}"
	}

	json_req('Post', "#{$testray_path}/runs.json", request_data, expected_status: expected_status)
	unless expected_status == 400
		@id = @hash_response['data']['testrayRunId']
		@run_num = @hash_response['data']['number']
	end
	self
end
