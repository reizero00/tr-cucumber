def merge_with(merge_partners, user_override: '')
    if merge_partners.class == Array # If arr of strings
        merge_str = "#{@name},#{merge_partners.join(',')}"
        json_req('Get', "#{$testray_path}/subtasks/merge.json", {}, uri_param: "?testrayTaskId=#{@parent_task.id}&names=#{merge_str}", expected_status: 200, username: user_override)
        @hash_response['data']['mergedWith'].each do |merged_name|
            $master_map['tasks'][@parent_task.name].subtasks.delete(merged_name)
        end
    elsif merge_partners.class == String
        merge_str = "#{@name},#{merge_partners.gsub(', ', ',')}"
        json_req('Get', "#{$testray_path}/subtasks/merge.json", {}, uri_param: "?testrayTaskId=#{@parent_task.id}&names=#{merge_str}", expected_status: 200, username: user_override)
    end
    self.refresh
end

def refresh
    json_req('Get', "#{$testray_path}/subtasks.json", {}, uri_param: "?testrayTaskId=#{@parent_task.id}", expected_status: 200)
    @hash_response['data'].each do |st_hash| # Update local map ST's to match index
        @parent_task.subtasks[st_hash['name']].setup(st_hash, @parent_task)
    end
    self
end

def setup(subtask_hash, parent_task)
    @parent_task = parent_task
    @name = subtask_hash['name']
    @id = subtask_hash['testraySubtaskId']
    @assignee = subtask_hash['assignedUser']['userName']
    @assignee_id = subtask_hash['assignedUser']['userId']
    @status = subtask_hash['statusLabel']
    @comment = subtask_hash['comment']
    @issues = subtask_hash['issues']
    @merged_with = subtask_hash['mergedWith']
    @split_from_children = subtask_hash['splitFromChildren']
    @split_from_parent = subtask_hash['splitFromParent']
    @n_tests = subtask_hash['tests']
    @score = subtask_hash['score']

    unless subtask_hash['testrayTaskId'] == parent_task.id # Ensure parent subtask matches
        raise("Mismatch between the task ID in the subtask_hash (#{subtask_hash['testrayTaskId']}) and the specified parent_task (#{parent_task.id})")
    end
    self
end

def split(case_result_ids, user_override: '')
    json_req('Get',"#{$testray_path}/subtasks/split.json?testrayTaskId=#{@parent_task.id}&name=#{@name}&testrayCaseResultIds=#{case_result_ids}", expected_status: 200, username: user_override) #returns new ST

    @parent_task.subtasks[@hash_response['data']['name']] = TestraySubtask.new(@hash_response['data'], @parent_task) # Make object for newly created ST from split
    self.refresh
end

def update_status(status, comment: '', issues: '', user_override: '')
    uri_params = "?id=#{@id}&statusLabel=#{status}"
    uri_params << "&comment=#{CGI::escape(comment)}" unless comment == nil
    uri_params << "&issues=#{issues}" unless issues == nil
    json_req('Get', "#{$testray_path}/subtasks/updateStatus.json", {}, uri_param: uri_params, expected_status: 200, username: user_override)
    self.refresh
end
