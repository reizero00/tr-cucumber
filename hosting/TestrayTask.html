def add(case_type_ids, user_ids, expected_status: 200)
    json_req('Get', "#{$testray_path}/tasks/add.json", {}, uri_param: "?name=#{CGI::escape(@name)}&testrayBuildId=#{@parent_build.id}&testrayCaseTypeIds=#{case_type_ids}&userIds=#{user_ids}", expected_status: expected_status)
    @id = @hash_response['data']['testrayTaskId']

    # Init vars with dummy vals
    task_loaded = nil
    complete_st = 0
    total_st = -1

    # Wait until the task has finished loading subtasks
    tries = 0
    until (task_loaded == 1 && complete_st/total_st == 1) || tries > 3
        sleep(5) unless tries == 0 || (task_loaded == 1 && (complete_st/total_st).nan?) # Do not sleep for the first attempt
        json_req('Get', "#{$testray_path}/tasks/progress.json", {}, uri_param: "?id=#{@id}", expected_status: 200)

        task_loaded = @hash_response['data']['completeTasks']
        complete_st = @hash_response['data']['completeSubtasks'].to_f
        total_st = @hash_response['data']['totalSubtasks'].to_f

        tries += 1
    end
    unless (task_loaded == 1 && complete_st/total_st == 1) || (complete_st/total_st).nan?
        raise("Unable to load the #{@name} task's subtasks within the allotted number of attempts. Task loaded status: #{task_loaded}. Loaded ST's: #{complete_st}. Total ST's: #{total_st}.")
    end

    # Create subtask objects
    json_req('Get', "#{$testray_path}/subtasks.json", {}, uri_param: "?testrayTaskId=#{@id}", expected_status: 200)
    @hash_response['data'].each do |subtask_hash|
        @subtasks[subtask_hash['name']] = TestraySubtask.new(subtask_hash, self)
    end
    self
end

def update(task_name: '', user_ids: '', expected_status: 200)
    uri_params = "?id=#{@id}"
    uri_params << "&name=#{CGI::escape(task_name)}" unless task_name == ''
    uri_params << "&userIds=#{user_ids}" unless user_ids == ''

    json_req('Get', "#{$testray_path}/tasks/update.json", {}, uri_param: uri_params, expected_status: expected_status)
    self
end
